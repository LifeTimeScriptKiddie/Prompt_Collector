def clean_and_format_query(query_text, truncate_len=3000):

    if query_text.startswith('"') and query_text.endswith('"'):
        query_text = query_text[1:-1]

        query_text = query_text[1:-1]

    # Remove leading/trailing whitespace
    query_text = query_text.strip()

    # Replace multiple newlines with a single newline, and escape pipe characters
    query_text = re.sub(r'\n+', r'\n', query_text).replace('|', '\|')

    # Truncate if longer than truncate_len characters
    if len(query_text) > truncate_len:
        query_text = query_text[:truncate_len - 3] + "..."

    # Ensure it's wrapped in backticks
    if not (query_text.startswith('`') and query_text.endswith('`')):
        # Handle cases where it might already be a multi-line code block
        if query_text.startswith('```') and query_text.endswith('```'):
            # Already a code block, keep as is
            pass
        else:
            query_text = f"`{query_text}`"
    return query_text
